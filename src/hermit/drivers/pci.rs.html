77	pub fn get_bar(&self, slot: u8) -> Option<Bar> {
78		let header = self.header();

86	/// Configure the bar at register `slot`
87	pub fn set_bar(&self, slot: u8, bar: Bar) {
88		let value = match bar {
89			Bar::Io { port } => (port | 1) as usize,
90			Bar::Memory32 {
91				address,
92				size: _,
93				prefetchable,
94			} => {
95				if prefetchable {
96					(address | (1 << 3)) as usize
97				} else {
98					address as usize
99				}
100			}
101			Bar::Memory64 {
102				address,
103				size: _,
104				prefetchable,
105			} => {
106				if prefetchable {
107					(address | (2 << 1) | (1 << 3)) as usize
108				} else {
109					(address | (2 << 1)) as usize
110				}
111			}
112		};
113		let mut header = EndpointHeader::from_header(self.header(), &self.access).unwrap();
114		unsafe {
115			header.write_bar(slot, &self.access, value).unwrap();
116		}
117	}

119	/// Memory maps pci bar with specified index to identical location in virtual memory.
120	/// no_cache determines if we set the `Cache Disable` flag in the page-table-entry.
121	/// Returns (virtual-pointer, size) if successful, else None (if bar non-existent or IOSpace)
122	pub fn memory_map_bar(&self, index: u8, no_cache: bool) -> Option<(VirtAddr, usize)> {
123		let (address, size, prefetchable, _width) = match self.get_bar(index) {
124			Some(Bar::Io { .. }) => {
125				warn!("Cannot map IOBar!");
126				return None;
127			}
128			Some(Bar::Memory32 {
129				address,
130				size,
131				prefetchable,
132			}) => (
133				u64::from(address),
134				usize::try_from(size).unwrap(),
135				prefetchable,
136				32,
137			),
138			Some(Bar::Memory64 {
139				address,
140				size,
141				prefetchable,
142			}) => (address, usize::try_from(size).unwrap(), prefetchable, 64),
143			_ => {
144				return None;
145			}
146		};
147
148		if address == 0 {
149			return None;
150		}
151
152		debug!("Mapping bar {index} at {address:#x} with length {size:#x}");
153
154		if !prefetchable {
155			warn!("Currently only mapping of prefetchable bars is supported!");
156		}
157
158		// Since the bios/bootloader manages the physical address space, the address got from the bar is unique and not overlapping.
159		// We therefore do not need to reserve any additional memory in our kernel.
160		// Map bar into RW^X virtual memory
161		let physical_address = address;
162		let virtual_address =
163			crate::mm::map(PhysAddr::new(physical_address), size, true, true, no_cache);
164
165		Some((virtual_address, size))
166	}

168	pub fn get_irq(&self) -> Option<InterruptLine> {
169		let header = self.header();
170		let endpoint = EndpointHeader::from_header(header, &self.access)?;
171		let (pin, line) = endpoint.interrupt(&self.access);
172		// PCIe specification v5 section 7.5.1.1.13 (Interrupt Pin Register)
173		match pin {
174			0 => {
175				warn!("The function uses no legacy interrupt message(s).");
176				None
177			}
178			1..=4 => {
179				// PCI specification v3 footnote 43
180				#[cfg(target_arch = "x86_64")]
181				if matches!(line, 16..254) {
182					error!("Reserved IRQ number");
183					return None;
184				} else if line == 255 {
185					error!("Unknown IRQ line or no connection to the interrupt controller");
186					return None;
187				}
188
189				Some(line)
190			}
191			5.. => {
192				error!("Reserved interrupt pin value returned.");
193				None
194			}
195		}
196	}

222impl<T: ConfigRegionAccess> fmt::Display for PciDevice<T> {
223	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
224		let header = self.header();
225		let header_type = header.header_type(&self.access);
226		let (vendor_id, device_id) = header.id(&self.access);
227		let (_dev_rev, class_id, subclass_id, _interface) = header.revision_and_class(&self.access);
228
229		if let Some(endpoint) = EndpointHeader::from_header(header, &self.access) {
230			#[cfg(feature = "pci-ids")]
231			let (class_name, vendor_name, device_name) = {
232				use pci_ids::{Class, Device, FromId, Subclass};
233
234				let class_name = Class::from_id(class_id).map_or("Unknown Class", |class| {
235					class
236						.subclasses()
237						.find(|s| s.id() == subclass_id)
238						.map(Subclass::name)
239						.unwrap_or_else(|| class.name())
240				});
241
242				let (vendor_name, device_name) = Device::from_vid_pid(vendor_id, device_id)
243					.map(|device| (device.vendor().name(), device.name()))
244					.unwrap_or(("Unknown Vendor", "Unknown Device"));
245
246				(class_name, vendor_name, device_name)
247			};
248
249			#[cfg(not(feature = "pci-ids"))]
250			let (class_name, vendor_name, device_name) =
251				("Unknown Class", "Unknown Vendor", "Unknown Device");
252
253			// Output detailed readable information about this device.
254			write!(
255				f,
256				"{:02X}:{:02X} {} [{:02X}{:02X}]: {} {} [{:04X}:{:04X}]",
257				self.address.bus(),
258				self.address.device(),
259				class_name,
260				class_id,
261				subclass_id,
262				vendor_name,
263				device_name,
264				vendor_id,
265				device_id
266			)?;
267
268			// If the devices uses an IRQ, output this one as well.
269			let (_, irq) = endpoint.interrupt(&self.access);
270			if irq != 0 && irq != u8::MAX {
271				write!(f, ", IRQ {irq}")?;
272			}
273
274			let mut slot: u8 = 0;
275			while usize::from(slot) < MAX_BARS {
276				if let Some(pci_bar) = endpoint.bar(slot, &self.access) {
277					match pci_bar {
278						Bar::Memory64 {
279							address,
280							size,
281							prefetchable,
282						} => {
283							write!(
284								f,
285								", BAR{slot} Memory64 {{ address: {address:#X}, size: {size:#X}, prefetchable: {prefetchable} }}"
286							)?;
287							slot += 1;
288						}
289						Bar::Memory32 {
290							address,
291							size,
292							prefetchable,
293						} => {
294							write!(
295								f,
296								", BAR{slot} Memory32 {{ address: {address:#X}, size: {size:#X}, prefetchable: {prefetchable} }}"
297							)?;
298						}
299						Bar::Io { port } => {
300							write!(f, ", BAR{slot} IO {{ port: {port:#X} }}")?;
301						}
302					}
303				}
304				slot += 1;
305			}
306		} else {
307			// Output detailed readable information about this device.
308			write!(
309				f,
310				"{:02X}:{:02X} {:?} [{:04X}:{:04X}]",
311				self.address.bus(),
312				self.address.device(),
313				header_type,
314				vendor_id,
315				device_id
316			)?;
317		}
318
319		Ok(())
320	}
321}

345impl PciDriver {
346	#[cfg(feature = "virtio-console")]
347	fn get_console_driver(&self) -> Option<&InterruptTicketMutex<VirtioConsoleDriver>> {
348		#[allow(unreachable_patterns)]
349		match self {
350			Self::VirtioConsole(drv) => Some(drv),
351			_ => None,
352		}
353	}
354
355	#[cfg(feature = "virtio-vsock")]
356	fn get_vsock_driver(&self) -> Option<&InterruptTicketMutex<VirtioVsockDriver>> {
357		#[allow(unreachable_patterns)]
358		match self {
359			Self::VirtioVsock(drv) => Some(drv),
360			_ => None,
361		}
362	}
363
364	#[cfg(feature = "virtio-fs")]
365	fn get_filesystem_driver(&self) -> Option<&InterruptTicketMutex<VirtioFsDriver>> {
366		match self {
367			Self::VirtioFs(drv) => Some(drv),
368			#[allow(unreachable_patterns)]
369			_ => None,
370		}
371	}
372
373	fn get_interrupt_handler(&self) -> (InterruptLine, fn()) {
374		#[allow(unreachable_patterns)]
375		match self {
376			#[cfg(feature = "virtio-vsock")]
377			Self::VirtioVsock(drv) => {
378				fn vsock_handler() {
379					if let Some(driver) = get_vsock_driver() {
380						driver.lock().handle_interrupt();
381					}
382				}
383
384				let irq_number = drv.lock().get_interrupt_number();
385
386				(irq_number, vsock_handler)
387			}
388			#[cfg(feature = "virtio-fs")]
389			Self::VirtioFs(drv) => {
390				fn fuse_handler() {}
391
392				let irq_number = drv.lock().get_interrupt_number();
393
394				(irq_number, fuse_handler)
395			}
396			#[cfg(feature = "virtio-console")]
397			Self::VirtioConsole(drv) => {
398				fn console_handler() {
399					if let Some(driver) = get_console_driver() {
400						driver.lock().handle_interrupt();
401					}
402				}
403
404				let irq_number = drv.lock().get_interrupt_number();
405				(irq_number, console_handler)
406			}
407			_ => todo!(),
408		}
409	}
410}

416pub(crate) fn get_interrupt_handlers() -> HashMap<InterruptLine, InterruptHandlerQueue, RandomState>
417{
418	let mut handlers: HashMap<InterruptLine, InterruptHandlerQueue, RandomState> =
419		HashMap::with_hasher(RandomState::with_seeds(0, 0, 0, 0));
420
421	for drv in PCI_DRIVERS.finalize().iter() {
422		let (irq_number, handler) = drv.get_interrupt_handler();
423
424		if let Some(map) = handlers.get_mut(&irq_number) {
425			map.push_back(handler);
426		} else {
427			let mut map: InterruptHandlerQueue = VecDeque::new();
428			map.push_back(handler);
429			handlers.insert(irq_number, map);
430		}
431	}
432
433	#[cfg(target_arch = "x86_64")]
434	{
435		use crate::kernel::serial::get_serial_handler;
436		let (irq_number, handler) = get_serial_handler();
437
438		if let Some(map) = handlers.get_mut(&irq_number) {
439			map.push_back(handler);
440		} else {
441			let mut map: InterruptHandlerQueue = VecDeque::new();
442			map.push_back(handler);
443			handlers.insert(irq_number, map);
444		}
445	}
446
447	#[cfg(any(feature = "rtl8139", feature = "virtio-net"))]
448	if let Some(device) = NETWORK_DEVICE.lock().as_ref() {
449		handlers
450			.entry(device.get_interrupt_number())
451			.or_default()
452			.push_back(crate::executor::network::network_handler);
453	}
454
455	handlers
456}

488pub(crate) fn init() {
489	// virtio: 4.1.2 PCI Device Discovery
490	without_interrupts(|| {
491		for adapter in PCI_DEVICES.finalize().iter().filter(|x| {
492			let (vendor_id, device_id) = x.id();
493			vendor_id == 0x1af4 && (0x1000..=0x107f).contains(&device_id)
494		}) {
495			info!(
496				"Found virtio device with device id {:#x}",
497				adapter.device_id()
498			);
499
500			#[cfg(feature = "virtio")]
501			match pci_virtio::init_device(adapter) {
502				#[cfg(feature = "virtio-console")]
503				Ok(VirtioDriver::Console(drv)) => {
504					register_driver(PciDriver::VirtioConsole(InterruptTicketMutex::new(*drv)));
505					info!("Switch to virtio console");
506					crate::console::CONSOLE
507						.lock()
508						.replace_device(IoDevice::Virtio(VirtioUART::new()));
509				}
510				#[cfg(feature = "virtio-fs")]
511				Ok(VirtioDriver::Fs(drv)) => {
512					register_driver(PciDriver::VirtioFs(InterruptTicketMutex::new(*drv)));
513				}
514				#[cfg(all(not(feature = "rtl8139"), feature = "virtio-net"))]
515				Ok(VirtioDriver::Net(drv)) => *crate::executor::device::NETWORK_DEVICE.lock() = Some(*drv),
516				#[cfg(feature = "virtio-vsock")]
517				Ok(VirtioDriver::Vsock(drv)) => {
518					register_driver(PciDriver::VirtioVsock(InterruptTicketMutex::new(*drv)));
519				}
520				Err(err) => error!("Could not initialize virtio-pci device: {err}"),
521			}
522		}
523
524		// Searching for Realtek RTL8139, which is supported by Qemu
525		#[cfg(feature = "rtl8139")]
526		for adapter in PCI_DEVICES.finalize().iter().filter(|x| {
527			let (vendor_id, device_id) = x.id();
528			vendor_id == 0x10ec && (0x8138..=0x8139).contains(&device_id)
529		}) {
530			info!(
531				"Found Realtek network device with device id {:#x}",
532				adapter.device_id()
533			);
534
535			match rtl8139::init_device(adapter) {
536				Ok(drv) => *crate::executor::device::NETWORK_DEVICE.lock() = Some(drv),
537				Err(err) => error!("Could not initialize rtl8139 device: {err}"),
538			}
539		}
540	});
